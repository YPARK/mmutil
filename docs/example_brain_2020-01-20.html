brain.colors =read_tsv("brain_colors.txt")
## Parsed with column specification:
## cols(
##   broad.cell.type = col_character(),
##   celltype = col_character(),
##   hex = col_character()
## )
.show.scatter = function(.argmax, .embed) {

    argmax.tab = fread(.argmax) %>%
        (function(x) { colnames(x) = c("TAG", "k"); x })

    embedded.tab = fread(.embed) %>%
        (function(x) { colnames(x) = c("e1", "e2"); x })

    ## Show 2D embedding of those clusters of cells

    .K = max(argmax.tab$k)
    .lab = c("cluster" %&% 1:.K, "unassigned")

    argmax.tab = argmax.tab[, k := factor(k, c(1:.K, -1), .lab)]

    .dt = cbind(argmax.tab, embedded.tab) %>%
        na.omit()

    .dt.sub = copy(.dt[, .SD[sample(.N, min(1000, .N))], by = k])

    p1 =
        .gg.plot(.dt.sub, aes(x=e1, y=e2, color=as.factor(k))) +
        geom_point(size=.5) +
        theme(legend.title = element_blank()) +
        theme(legend.position=c(0,0), legend.justification=c(0,0))

    print(p1)

    ## Remove outlier cells
    .dt[, c("barcode", "projid", "idx") := tstrsplit(TAG, "_", fixed=TRUE)]
    .dt[, projid := as.numeric(projid)]

    .df = meta.tab %>%
        separate("TAG", c("barcode", "batch"), sep="[.]") %>%
        select(-batch) %>% 
        left_join(.dt) %>%
        na.omit()

    .color = .df %>%
        select(broad.cell.type) %>%
        distinct() %>% 
        arrange(broad.cell.type) %>%
        left_join(brain.colors)

    .df = .df %>%
        mutate(broad.cell.type = factor(broad.cell.type, .color$broad.cell.type, .color$celltype)) %>%
        arrange(broad.cell.type)

    p2 =
        .gg.plot(.df, aes(x=e1, y=e2, color=broad.cell.type)) +
        geom_point(size=.5) +
        scale_color_manual(values = .color$hex) +
        theme(legend.title = element_blank()) +
        theme(legend.position=c(0,0), legend.justification=c(0,0))

    print(p2)
}
